\begin{tikzpicture}[scale=1.2]
    % Draw coordinate system
    \draw[axis] (-3,0) -- (3,0) node[right] {$x$};
    \draw[axis] (0,-2) -- (0,3) node[above] {$y$};
    
    % Add grid
    \draw[grid, step=0.5] (-3,-2) grid (3,3);
    
    % Draw unit circle
    \draw[thick, primaryblue] (0,0) circle (1);
    
    % Mark important angles
    \foreach \angle in {0,30,45,60,90,120,135,150,180,210,225,240,270,300,315,330}
    {
        \draw (0,0) -- (\angle:1.2);
        \fill[point] (\angle:1) circle (1pt);
    }
    
    % Label some key points
    \node[right] at (1,0) {$(1,0)$};
    \node[above] at (0,1) {$(0,1)$};
    \node[left] at (-1,0) {$(-1,0)$};
    \node[below] at (0,-1) {$(0,-1)$};
    
    % Label the circle
    \node[primaryblue] at (1.5,1.5) {Unit Circle};
    
    % Add some trigonometric values
    \node[above right] at (0.707,0.707) {$(\frac{\sqrt{2}}{2}, \frac{\sqrt{2}}{2})$};
    \node[below right] at (0.5,-0.866) {$(\frac{1}{2}, -\frac{\sqrt{3}}{2})$};
\end{tikzpicture}
